Program ->
    DefList Start |
    Start
 .

DefList ->
    Def DefList |
    Def  .

Def ->
    Class |
    Impl  .

Start ->
    start MethodBlock .

Class ->
    class idclass InheritanceOpt bra_o AttributeListOpt bra_c  .

InheritanceOpt ->
    Inheritance |
      .

Inheritance -> colon Type .

AttributeListOpt ->
    AttributeList |
      .

AttributeList ->
    Attribute AttributeList |
    Attribute .

Attribute -> VisibilityOpt Type VarsDeclList semi .

VisibilityOpt ->
    Visibility |
      .

Visibility -> pub .

Impl -> impl idclass bra_o MemberListOpt bra_c .

MemberListOpt ->
    MemberList |
      .

MemberList ->
    Member MemberList |
    Member .

Member ->
    Method |
    Constructor .

Constructor -> dot FormalArgs MethodBlock .

Method ->
    MethodForm fn MethodSignatureAndBody |
    fn MethodSignatureAndBody .

MethodForm -> st .

MethodSignatureAndBody -> MethodTypeOpt idMetAt FormalArgs MethodBlock .

MethodTypeOpt ->
    MethodType |
      .

MethodType ->
    Type |
    void .

MethodBlock -> bra_o LocalVarsDeclListOpt SentenceListOpt bra_c .

LocalVarsDeclListOpt ->
    LocalVarsDeclList |
      .

LocalVarsDeclList ->
    LocalVarsDecl LocalVarsDeclList |
    LocalVarsDecl .

LocalVarsDecl -> Type VarsDeclList semi .

VarsDeclList -> idMetAt VarsDeclTail .

VarsDeclTail ->
    comma idMetAt VarsDeclTail |
      .

FormalArgs -> par_o FormalArgsListOpt par_c .

FormalArgsListOpt ->
    FormalArgsList |
      .

FormalArgsList -> FormalArg FormalArgsTail .

FormalArgsTail ->
    comma FormalArg FormalArgsTail |
      .

FormalArg -> Type idMetAt .

Type ->
    ArrayType
    PrimitiveType |
    ReferenceType | .

PrimitiveType ->
    int |
    bool |
    str .

ReferenceType -> idclass .

ArrayType -> array PrimitiveType .

SentenceListOpt ->
    SentenceList |
      .

SentenceList ->
    Sentence SentenceList |
    Sentence .

Sentence ->
    semi |
    Assignment semi |
    SimpleSentence semi |
    if par_o Exp par_c Sentence ElseOpt |
    while par_o Exp par_c Sentence |
    for par_o PrimitiveType idMetAt in idMetAt par_c Sentence |
    Block |
    ret RetOpt semi .

ElseOpt ->
    else Sentence |
      .

RetOpt ->
    Exp |
      .

Block -> bra_o SentenceListOpt bra_c .

Assignment -> Assignable eq Exp .

Assignable ->
    self SelfChainOpt |
    idMetAt AssignableTail .

SelfChainOpt ->
    dot idMetAt SelfChainOpt |
      .

AssignableTail ->
    dot idMetAt AssignableTail |
    sqr_o Exp sqr_c AssignableTail |
      .

SimpleSentence -> par_o Exp par_c .

Exp -> OrExp .

OrExp -> AndExp OrExpTail .

OrExpTail ->
    or AndExp OrExpTail |
      .

AndExp -> EqExp AndExpTail .

AndExpTail ->
    and EqExp AndExpTail |
      .

EqExp -> CompoundExp EqExpTail .

EqExpTail ->
    EqOp CompoundExp EqExpTail |
      .

EqOp ->
    eqeq |
    ineq .

CompoundExp -> AddExp CompoundExpTail .

CompoundExpTail ->
    CompoundOp AddExp CompoundExpTail |
      .

CompoundOp ->
    less |
    great |
    lesseq |
    greateq .

AddExp -> MultExp AddExpTail .

AddExpTail ->
    AddOp MultExp AddExpTail |
      .

AddOp ->
    plus |
    minus .

MultExp -> UnaryExp MultExpTail .

MultExpTail ->
    MultOp UnaryExp MultExpTail |
      .

MultOp ->
    mult |
    div |
    mod .

UnaryExp ->
    UnaryOp UnaryExp |
    Operand .

UnaryOp ->
    plus |
    minus |
    not .

Operand ->
    Literal |
    Primary .

Literal ->
    intliteral |
    true |
    false |
    strliteral |
    nil .

Primary ->
    BracketedExp OptionalChaining |
    self OptionalChaining |
    NewExpr |
    StaticCallOrRef |
    IdPrimary .

BracketedExp -> par_o Exp par_c .

IdPrimary -> idMetAt IdPrimaryTail OptionalChaining .

IdPrimaryTail ->
    CurrentArguments |
    sqr_o Exp sqr_c |
      .

StaticCallOrRef -> idclass StaticCallTail OptionalChaining .

StaticCallTail ->
    dot idMetAt CurrentArguments |
      .

NewExpr -> new NewTail .

NewTail ->
    PrimitiveType sqr_o Exp sqr_c |
    idclass CurrentArguments OptionalChaining .

CurrentArguments -> par_o ExpsListOpt par_c .

ExpsListOpt ->
    ExpsList |
      .

ExpsList -> Exp ExpsTail .

ExpsTail ->
    comma Exp ExpsTail |
      .

OptionalChaining ->
    dot idMetAt ChainAccess OptionalChaining |
      .

ChainAccess ->
    CurrentArguments |
    sqr_o Exp sqr_c |.
