Program ->
    DefsList Start |
    Start

DefsList ->
    Def DefsList |
    Def

Def ->
    Class |
    Impl

Start ->
    start MethodBlock

Class ->
    class idclass : Type { AttributeRep } |
    class idclass : Type { } |
    class idclass { AttributeRep } |
    class idclass { }

AttributeRep ->
    Attribute AttributeRep |
    Attribute

Attribute ->
    pub Type VarsDeclList ; |
    Type VarsDeclList ;

Impl ->
    impl idclass { MemberRep } |
    impl idclass { }

MemberRep ->
    Member MemberRep |
    Member

Member ->
    Constructor |
    Method

Constructor ->
    . FormalArgs MethodBlock

Method ->
    st fn MethodSignatureAndBody |
    fn MethodSignatureAndBody

MethodSignatureAndBody ->
    MethodType idMetAt FormalArgs MethodBlock |
    idMetAt FormalArgs MethodBlock

MethodType ->
    Type |
    void

FormalArgs ->
    ( FormalArgsList ) |
    ( )

FormalArgsList ->
    FormalArg FormalArgsTail

FormalArgsTail ->
    , FormalArg FormalArgsTail |
    λ

FormalArg ->
    Type idMetAt

MethodBlock ->
    { LocalVarsDeclRep SentenceRep } |
    { LocalVarsDeclRep } |
    { SentenceRep } |
    { }

LocalVarsDeclRep ->
    LocalVarsDecl LocalVarsDeclRep |
    LocalVarsDecl

LocalVarsDecl ->
    Type VarsDeclList ;

VarsDeclList ->
    idMetAt VarsDeclTail

VarsDeclTail ->
    , idMetAt VarsDeclTail |
    λ

Type ->
    PrimitiveType |
    ReferenceType |
    ArrayType

PrimitiveType ->
    _Int |
    _Bool |
    _Str

ReferenceType ->
    idclass

ArrayType ->
    _Array PrimitiveType

SentenceRep ->
    Sentence SentenceRep |
    Sentence

Sentence ->
    ; |
    if ( Exp ) Sentence else Sentence |
    if ( Exp ) Sentence |
    while ( Exp ) Sentence |
    for ( PrimitiveType idMetAt in idMetAt ) Sentence |
    Block |
    ret Exp ; |
    ret ; |
    SimpleSentence ; |
    Assignment ;

Block ->
    { SentenceRep } |
    { }

SimpleSentence ->
    ( Exp )

Assignment ->
    self SelfAssignableTail = Exp |
    idMetAt IdAssignableTail = Exp

SelfAssignableTail ->
    . idMetAt SelfAssignableTail |
    λ

IdAssignableTail ->
    . idMetAt IdAssignableTail |
    [ Exp ] IdAssignableTail |
    λ

Exp -> OrExp

OrExp ->
    AndExp OrExpTail

OrExpTail ->
    or AndExp OrExpTail |
    λ

AndExp ->
    EqExp AndExpTail

AndExpTail ->
    and EqExp AndExpTail |
    λ

EqExp -> CompoundExp EqExpTail

EqExpTail ->
    EqOp CompoundExp EqExpTail |
    λ

EqOp ->
    == |
    !=

CompoundExp ->
    AddExp CompoundExpTail

CompoundExpTail ->
    CompoundOp AddExp CompoundExpTail |
    λ

CompoundOp ->
    < |
    > |
    <= |
    >=

AddExp ->
    MultExp AddExpTail

AddExpTail ->
    AddOp MultExp AddExpTail |
    λ

AddOp ->
    + |
    -

MultExp ->
    UnaryExp MultExpTail

MultExpTail ->
    MultOp UnaryExp MultExpTail |
    λ

MultOp ->
    * |
    / |
    %

UnaryExp ->
    UnaryOp UnaryExp |
    Operand

UnaryOp ->
    + |
    - |
    !

Operand ->
    Literal |
    Primary

Literal ->
    intLiteral |
    true |
    false |
    _StrLiteral |
    nil

Primary ->
    ( Exp ) OptionalChaining |
    self OptionalChaining |
    NewExpr |
    StaticCallOrRef |
    IdPrimary

IdPrimary ->
    idMetAt IdPrimaryTail OptionalChaining

IdPrimaryTail ->
    CurrentArguments |
    [ Exp ] |
    λ

StaticCallOrRef ->
    idclass StaticCallTail OptionalChaining

StaticCallTail ->
    . idMetAt CurrentArguments |
    λ

NewExpr ->
    new PrimitiveType [ Exp ] |
    new idclass CurrentArguments OptionalChaining

CurrentArguments ->
    ( ExpsList ) |
    ( )

ExpsList -> Exp ExpsTail

ExpsTail ->
    , Exp ExpsTail |
    λ

OptionalChaining ->
    . idMetAt ChainAccess OptionalChaining |
    λ

ChainAccess ->
    CurrentArguments |
    [ Exp ] |
    λ
