Program ::=
    DefsList Start

Start ::=
    start MethodBlock

DefsList ::=
    Class DefsList |
    Impl DefsList |
    #

Class ::=
    class idclass Inheritance? { Attribute∗ }

Impl ::=
    impl idclass { Member∗ }

Inheritance ::=
    : Type

Member ::=
    Method |
    Constructor

Constructor ::=
    . FormalArgs MethodBlock

Attribute ::=
    Visibility? Type VarsDeclList ;

Method ::=
    MethodForm? fn MethodType? idMetAt FormalArgs MethodBlock

Visibility ::=
    pub

MethodForm ::=
    st

MethodBlock ::=
    { LocalVarsDecl∗ Sentence∗ }

LocalVarsDecl ::=
    Type VarsDeclList ;

VarsDeclList ::=
    idMetAt |
    idMetAt , VarsDeclList

FormalArgs ::=
    ( FormalArgsList? )

FormalArgsList ::=
    FormalArg , FormalArgsList |
    FormalArg

FormalArg ::=
    Type idMetAt

MethodType ::=
    Type |
    void

Type ::=
    PrimitiveType |
    ReferenceType |
    ArrayType

PrimitiveType ::=
    str |
    bool |
    int

ReferenceType ::=
    idclass

ArrayType ::=
    array PrimitiveType

Sentence ::=
    ; |
    Assignment ; |
    SimpleSentence ; |
    if ( Exp ) Sentence |
    if ( Exp ) Sentence else Sentence |
    while ( Exp ) Sentence |
    for ( PrimitiveType idMetAt in idMetAt ) Sentence |
    Block |
    ret Exp? ;

Block ::=
    { Sentence∗ }

Assignment ::=
    SimpleVarAccess = Exp |
    SimpleSelfAccess = Exp

SimpleVarAccess ::=
    id SimpleChaining∗ |
    id [ Exp ]

SimpleSelfAccess ::=
    self SimpleChaining∗

SimpleChaining ::=
    . id

SimpleSentence ::=
    ( Exp )

Exp ::=
    OrExp

OrExp ::=
    OrExp \\ AndExp |
    AndExp

AndExp ::=
    AndExp && EqExp |
    EqExp

EqExp ::=
    EqExp EqOp CompoundExp |
    CompoundExp

CompoundExp ::=
    AdExp CompoundOp AdExp |
    AdExp

AdExp ::=
    AdExp AddOp MultExp |
    MultExp

MultExp ::=
    MultExp MultOp UnaryExp |
    UnaryExp

UnaryExp ::=
    UnaryOp UnaryExp |
    Operand

EqOp ::=
    == |
    !=

CompoundOp ::=
    < |
    > |
    <= |
    >=

AddOp ::=
    + |
    -

UnaryOp ::=
    + |
    - |
    ! |
    ++ |
    -- |
    (Int)

MultOp ::=
    * |
    /

Operand ::=
    Literal |
    Primary Chaining?

Literal ::=
    nil |
    true |
    false |
    intLiteral |
    strLiteral

Primary ::=
    BracketedExp |
    SelfAccess |
    VarAccess |
    MethodCall |
    StaticMethodCall |
    ClassConstructorCall

BracketedExp ::=
    ( Exp ) Chaining?

SelfAccess ::=
    self Chaining?

VarAccess ::=
    id Chaining? |
    id [ Exp ] Chaining?

MethodCall ::=
    id CurrentArgs Chaining?

StaticMethodCall ::=
    idclass . MethodCall Chaining?

ClassConstructorCall ::=
    new idclass CurrentArgs Chaining? |
    new PrimitiveType [ Exp ]

CurrentArgs ::=
    ( ExpsList? )

ExpsList ::=
    Exp |
    Exp , ExpsList

Chaining ::=
    . ChainedMethodCall |
    . ChainedVarAccess

ChainedMethodCall ::=
    id CurrentArgs Chaining?

ChainedVarAccess ::=
    id Chaining? |
    id [ Exp ] Chaining?
