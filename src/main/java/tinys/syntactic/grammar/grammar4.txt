Program ::=
    DefList Start


DefList ::=
    Def DefList |
    λ

Def ::=
    Class |
    Impl

Start ::=
    start MethodBlock

Class ::=
    class idclass InheritanceOpt { AttributeList }

InheritanceOpt ::=
    : Type |
    λ

AttributeList ::=
    Attribute AttributeList |
    λ

Attribute ::=
    VisibilityOpt Type idMetAt VarDeclList ;

VisibilityOpt ::=
    pub |
    λ

Impl ::=
    impl idclass { MemberList }

MemberList ::=
    Member MemberList |
    λ

Member ::=
    Method |
    Constructor

Constructor ::=
    . FormalArgs MethodBlock

Method ::=
    MethodFormOpt fn MethodSignatureAndBody

MethodFormOpt ::=
    st |
    λ

MethodSignatureAndBody ::=
    MethodTypeOpt idMetAt FormalArgs MethodBlock

MethodTypeOpt ::=
    Type |
    void |
    λ

MethodBlock ::=
    { LocalVarDeclList SentenceList }

LocalVarDeclList ::=
    LocalVarDecl LocalVarDeclList |
    λ

LocalVarDecl ::=
    Type idMetAt VarDeclList ;

VarDeclList ::=
    idMetAt VarDeclTail |
    λ

VarDeclTail ::=
    , idMetAt VarDeclTail |
    λ

FormalArgs ::=
    ( FormalArgsList )

FormalArgsList ::=
    FormalArg FormalArgsTail |
    λ

FormalArgsTail ::=
    , FormalArg FormalArgsTail |
    λ

FormalArg ::= 
    Type idMetAt

Type ::=
    ArrayType
    PrimitiveType |
    ReferenceType

ArrayType ::=
    array PrimitiveType

PrimitiveType ::=
    int |
    bool |
    str

ReferenceType ::=
    idclass

SentenceList ::=
    Sentence SentenceList |
    λ

Sentence ::=
    ; |
    Assignment ; |
    SimpleSentence ; |
    if ( Exp ) Sentence ElseOpt |
    while ( Exp ) Sentence |
    for ( PrimitiveType idMetAt in idMetAt ) Sentence |
    Block |
    ret RetOpt ;

ElseOpt ::=
    else Sentence |
    λ

RetOpt ::=
    Exp |
    λ

Block ::= 
    { SentenceList }

Assignment ::= 
    Assignable = Exp

Assignable ::=
    idMetAt VarAccessTail
    self SelfAccessTail |

SelfAccessTail ::=
    . idMetAt SelfAccessTail |
    λ

VarAccessTail ::=
    . idMetAt VarAccessTail |
    [ Exp ] VarAccessTail |
    λ

SimpleSentence ::=
    ( Exp )

Exp ::= 
    OrExp

OrExp ::= 
    AndExp OrExpTail

OrExpTail ::=
    \\ AndExp OrExpTail |
    λ

AndExp ::= 
    EqExp AndExpTail

AndExpTail ::=
    && EqExp AndExpTail |
    λ

EqExp ::= 
    CompoundExp EqExpTail

EqExpTail ::=
    EqOp CompoundExp EqExpTail |
    λ

CompoundExp ::=
    AddExp CompoundExpTail

CompoundExpTail ::=
    CompoundOp AddExp CompoundExpTail |
    λ

AddExp ::= 
    MultExp AddExpTail

AddExpTail ::=
    AddOp MultExp AddExpTail |
    λ

MultExp ::=
    UnaryExp MultExpTail

MultExpTail ::=
    MultOp UnaryExp MultExpTail |
    λ

UnaryExp ::=
    UnaryOp UnaryExp |
    Operand

EqOp ::=
    == |
    !=

CompoundOp ::=
    < |
    > |
    <= |
    >=

AddOp ::=
    + |
    -

MultOp ::=
    * |
    /


UnaryOp ::=
    + |
    - |
    ! |
    ++ |
    --

Operand ::=
    Literal |
    Primary

Primary ::=
    BracketedExp |
    SelfAccess |
    StaticMethodCall |
    ClassConstructorCall |
    IdPrimary

BracketedExp ::=
    ( Exp ) ChainingOpt

SelfAccess ::=
    self ChainingOpt

IdPrimary ::=
    idMetAt IdPrimaryTail ChainingOpt

IdPrimaryTail ::=
    CurrentArguments |
    [ Exp ] |
    λ


StaticMethodCall ::=
    idclass . idMetAt CurrentArguments ChainingOpt

ClassConstructorCall ::=
    new ClassConstructorCallTail

ClassConstructorCallTail ::=
    idclass CurrentArguments ChainingOpt |
    PrimitiveType [ Exp ]

CurrentArguments ::=
    ( ExpsList )

ExpsList ::=
    Exp ExpsTail |
    λ

ExpsTail ::=
    , Exp ExpsTail |
    λ

ChainingOpt ::=
    . idMetAt ChainBindOpt ChainingOpt |
    λ

ChainBindOpt ::=
    CurrentArguments |
    [ Exp ] |
    λ
