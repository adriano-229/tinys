Program ::=
    DefList Start


DefList ::=
    Def DefList |
    λ

Def ::=
    Class |
    Impl

Start ::=
    start MethodBlock

Class ::=
    class idclass InheritanceOpt { AttributeList }

InheritanceOpt ::=
    : Type |
    λ

AttributeList ::=
    Attribute AttributeList |
    λ

Attribute ::=
    VisibilityOpt Type VarsDeclList ;

VisibilityOpt ::=
    pub |
    λ

Impl ::=
    impl idclass { MemberList }

MemberList ::=
    Member MemberList |
    λ

Member ::=
    Method |
    Constructor

Constructor ::=
    . FormalArgs MethodBlock

Method ::=
    MethodFormOpt fn MethodSignatureAndBody

MethodFormOpt ::=
    st |
    λ

MethodSignatureAndBody ::=
    MethodTypeOpt idMetAt FormalArgs MethodBlock

MethodTypeOpt ::=
    Type |
    void |
    λ

MethodBlock ::=
    { LocalVarsDeclList SentenceListOpt }

LocalVarsDeclList ::=
    LocalVarsDecl LocalVarsDeclList |
    λ

LocalVarsDecl ::=
    Type VarsDeclList ;

VarsDeclList ::=
    idMetAt VarsDeclTail

VarsDeclTail ::=
    , idMetAt VarsDeclTail |
    λ

FormalArgs ::= ( FormalArgsListOpt )

FormalArgsListOpt ::=
    FormalArgsList |
    λ

FormalArgsList ::= FormalArg FormalArgsTail

FormalArgsTail ::=
    , FormalArg FormalArgsTail |
    λ

FormalArg ::= Type idMetAt

Type ::=
    ArrayType
    PrimitiveType |
    ReferenceType |

ArrayType ::=
    array PrimitiveType

PrimitiveType ::=
    int |
    bool |
    str

ReferenceType ::=
    idclass

SentenceList ::=
    Sentence SentenceList |
    λ

Sentence ::=
    ; |
    Assignment ; |
    SimpleSentence ; |
    if ( Exp ) Sentence ElseOpt |
    while ( Exp ) Sentence |
    for ( PrimitiveType idMetAt in idMetAt ) Sentence |
    Block |
    ret RetOpt ;

ElseOpt ::=
    else Sentence |
    λ

RetOpt ::=
    Exp |
    λ

Block ::= { SentenceListOpt }

Assignment ::= Assignable = Exp

Assignable ::=
    self SelfChainOpt |
    idMetAt AssignableTail

SelfChainOpt ::=
    . idMetAt SelfChainOpt |
    λ

AssignableTail ::=
    . idMetAt AssignableTail |
    [ Exp ] AssignableTail |
    λ

SimpleSentence ::= ( Exp )

Exp ::= OrExp

OrExp ::= AndExp OrExpTail

OrExpTail ::=
    or AndExp OrExpTail |
    λ

AndExp ::= EqExp AndExpTail

AndExpTail ::=
    and EqExp AndExpTail |
    λ

EqExp ::= CompoundExp EqExpTail

EqExpTail ::=
    EqOp CompoundExp EqExpTail |
    λ

EqOp ::=
    == |
    !=

CompoundExp ::= AddExp CompoundExpTail

CompoundExpTail ::=
    CompoundOp AddExp CompoundExpTail |
    λ

CompoundOp ::=
    < |
    > |
    <= |
    >=

AddExp ::= MultExp AddExpTail

AddExpTail ::=
    AddOp MultExp AddExpTail |
    λ

AddOp ::=
    + |
    -

MultExp ::= UnaryExp MultExpTail

MultExpTail ::=
    MultOp UnaryExp MultExpTail |
    λ

MultOp ::=
    * |
    / |
    %

UnaryExp ::=
    UnaryOp UnaryExp |
    Operand

UnaryOp ::=
    + |
    - |
    !

Operand ::=
    Literal |
    Primary

Literal ::=
    intLiteral |
    true |
    false |
    _StrLiteral |
    nil

Primary ::=
    BracketedExp OptionalChaining |
    self OptionalChaining |
    NewExpr |
    StaticCallOrRef |
    IdPrimary

BracketedExp ::= ( Exp )

IdPrimary ::= idMetAt IdPrimaryTail OptionalChaining

IdPrimaryTail ::=
    CurrentArguments |
    [ Exp ] |
    λ

StaticCallOrRef ::= idclass StaticCallTail OptionalChaining

StaticCallTail ::=
    . idMetAt CurrentArguments |
    λ

NewExpr ::= new NewTail

NewTail ::=
    PrimitiveType [ Exp ] |
    idclass CurrentArguments OptionalChaining

CurrentArguments ::= ( ExpsListOpt )

ExpsListOpt ::=
    ExpsList |
    λ

ExpsList ::= Exp ExpsTail

ExpsTail ::=
    , Exp ExpsTail |
    λ

OptionalChaining ::=
    . idMetAt ChainAccess OptionalChaining |
    λ

ChainAccess ::=
    CurrentArguments |
    [ Exp ] |
    λ
